function n(v) {

function sumRows(rows) {
  const out = { absolute: 0, percent: 0 };
  for (const r of rows) {
    out.absolute += n(r.absolute);
    out.percent += n(r.percent);
  }
  return out;
}

function fmtSigned(v, fixed = 2) {
  const num = n(v);
  if (Math.abs(num) < 1e-9) return "0";
  const p = Number(num.toFixed(fixed));
  return `${p >= 0 ? "+" : ""}${p}`;
}

function fmtSignedPct(v, fixed = 2) {
  const p = n(v) * 100;
  if (Math.abs(p) < 1e-9) return "0%";
  return `${p >= 0 ? "+" : ""}${Number(p.toFixed(fixed))}%`;
}
function fmtPartEffect(effect) {
  const abs = n(effect?.absolute);
  const pct = n(effect?.percent);
  if (Math.abs(abs) > 1e-9 && Math.abs(pct) > 1e-9) return `${fmtSigned(abs)} / ${fmtSignedPct(pct)}`;
  if (Math.abs(abs) > 1e-9) return fmtSigned(abs);
  if (Math.abs(pct) > 1e-9) return fmtSignedPct(pct);
  return "0";
}

function fmtTraitNum(v) {
  return Number(n(v).toFixed(2)).toString();
}

function parseHiddenValues(raw) {
  const s = String(raw || "").trim();
  if (!s) return null;
  const limMatch = s.match(/Limit\s*=\s*(-?\d+)/i);
  if (!limMatch) return null;
  const p1Match = s.match(/Param1\s*=\s*(-?\d+(?:\.\d+)?)/i);
  return {
    limit: Number(limMatch[1]),
    param1: p1Match ? Number(p1Match[1]) : 0,
  };
}

function conditionTextFromLimit(limit, param1) {
  if (Number(limit) === 11) return "在比赛车 开始时";
  const tpl = LIMIT_TO_CONDITION_ZH[Number(limit)];
  if (!tpl) return "";
  let p = Number(param1);
  // `*_First*` mods are encoded in 0.5s units in source data (e.g. 16 => 8s).
  if (Number(limit) === 1 && Number.isFinite(p) && Math.abs(p) > 1e-9) p /= 2;
  return tpl.replace("{0}", fmtTraitNum(p));
}

function parseRequirementFromConditionText(rawText) {
  const text = String(rawText || "").trim();
  if (!text) return null;
  const numMatch = text.match(/大于等于\s*(-?\d+(?:\.\d+)?)/);
  if (!numMatch) return null;
  const threshold = Number(numMatch[1]);
  if (!Number.isFinite(threshold)) return null;
  for (const [alias, key] of REQUIREMENT_KEY_ALIASES) {
    if (text.includes(alias)) return { key, threshold, alias };
  }
  return null;
}

function isRuntimeOnlyConditionText(rawText) {
  const text = String(rawText || "").trim();
  if (!text) return false;
  if (parseRequirementFromConditionText(text)) return false;
  return RUNTIME_CONDITION_PATTERNS.some((re) => re.test(text));
}

function firstEffectText(trait) {
  const effects = Array.isArray(trait?.effects) ? trait.effects : [];
  const e = effects.find((it) => Math.abs(n(it?.absolute)) > 1e-9 || Math.abs(n(it?.percent)) > 1e-9) || effects[0];
  if (!e) return "";
  const key = String(e.key || "");
  const label = String(e.type_zh || e.labelZh || metaLabel(key) || "").trim();
  if (!label) return "";
  const abs = n(e.absolute);
  const pct = n(e.percent);
  if (Math.abs(pct) > 1e-9) return `${label}${fmtSignedPct(pct)}`;
  if (Math.abs(abs) > 1e-9) return `${label}${fmtSigned(abs)}`;
  return label;
}

function deriveTriggerText(trait) {
  const effects = Array.isArray(trait?.effects) ? trait.effects : [];
  const conditionFromEffect = effects.map((e) => String(e.condition_desc || e.conditionDesc || "").trim()).find((x) => x);
  if (conditionFromEffect) {
    const parsed = parseHiddenValues(conditionFromEffect);
    if (parsed) {
      const mapped = conditionTextFromLimit(parsed.limit, parsed.param1);
      if (mapped) return mapped;
    }
    return conditionFromEffect;
  }

  const hidden = parseHiddenValues(trait?.hiddenValues || trait?.hidden_values);
  if (hidden) {
    const fromLimit = conditionTextFromLimit(hidden.limit, hidden.param1);
    if (fromLimit) return fromLimit;
  }

  const fromBoostType = effects
    .map((e) => BOOST_TRIGGER_BY_TYPE_ZH[Number(e.modifierType)])
    .find((txt) => String(txt || "").trim());
  if (fromBoostType) return fromBoostType;

  const trigger = String(trait?.trigger || "").trim();
  return isPlaceholderText(trigger) ? "" : trigger;
}

function deriveDescriptionText(trait) {
  const allMainSummary = allMainSamePercentSummary(trait);
  const trigger = deriveTriggerText(trait);
  const primary = allMainSummary || firstEffectText(trait);
  if (trigger && primary) return `${trigger}，${primary}`;
  if (primary) return primary;
  if (trigger) return trigger;
  const raw = String(trait?.descriptionZh || trait?.description_zh || "").trim();
  return isPlaceholderText(raw) ? "" : raw;
}

function lerp(v0, v1, t) {
  return v0 + (v1 - v0) * t;
}

function traitNameZh(trait) {
  const effectRows = Array.isArray(trait?.effects) ? trait.effects : [];
  const mainPctRows = effectRows.filter((e) => MAIN_KEYS.includes(String(e?.key || "")) && Math.abs(n(e?.percent)) > 1e-9);
  const nonZeroRows = effectRows.filter((e) => Math.abs(n(e?.absolute)) > 1e-9 || Math.abs(n(e?.percent)) > 1e-9);
  if (
    mainPctRows.length >= 4 &&
    MAIN_KEYS.every((k) => mainPctRows.some((e) => String(e.key) === k)) &&
    mainPctRows.every((e) => Math.abs(n(e.absolute)) < 1e-9) &&
    nonZeroRows.length === mainPctRows.length
  ) {
    const basePct = n(mainPctRows[0]?.percent);
    const samePct = mainPctRows.every((e) => Math.abs(n(e.percent) - basePct) < 1e-9);
    if (samePct) return `主要属性${fmtSignedPct(basePct)}`;
  }

  const explicit = String(trait?.nameZh || trait?.name_zh || "").trim();
  if (explicit) return explicit;
  const desc = String(trait?.descriptionZh || trait?.description_zh || "").trim();
  if (desc.includes("，")) {
    const right = desc.split("，").slice(1).join("，").trim();
    if (right) return right;
  }
  const effects = String(trait?.effectsZh || trait?.effects_zh || "").trim();
  if (effects) {
    const firstSeg = effects.split(";")[0]?.trim() || effects;
    const stripped = firstSeg.replace(/^\[[^\]]+\]\s*/, "").trim();
    if (stripped) return stripped;
  }
  return String(trait?.name || "未命名特性");
}

function traitNameEn(trait) {
  const explicit = String(trait?.nameEn || trait?.name_en || "").trim();
  if (explicit) return explicit;
  const desc = String(trait?.descriptionEn || trait?.description_en || "").trim();
  if (desc.includes(",")) {
    const right = desc.split(",").slice(1).join(",").trim();
    if (right) return right;
  }
  const effects = String(trait?.effectsEn || trait?.effects_en || "").trim();
  if (effects) {
    const firstSeg = effects.split(";")[0]?.trim() || effects;
    const stripped = firstSeg.replace(/^\[[^\]]+\]\s*/, "").trim();
    if (stripped) return stripped;
  }
  return String(trait?.name || t("roster.detail.unnamedTrait"));
}

function normalizeSkinKeyForOwnership(raw) {
  let out = String(raw || "").trim().toUpperCase();
  if (!out) return "";
  out = out.replace(/[\s-]+/g, "_");
  out = out.replace(/_SKIN_?(\d+)/g, "_SK$1");
  out = out.replace(/SKIN_?(\d+)/g, "_SK$1");
  out = out.replace(/__+/g, "_");
  return out;
}

function expandSkinOwnershipKeys(raw, carId = "") {
  const out = new Set();
  const normalized = normalizeSkinKeyForOwnership(raw);
  if (!normalized) return out;
  out.add(normalized);
  const car = String(carId || "").trim().toUpperCase();
  const alreadyQualifiedSkinId = /^[A-Z0-9]+_SK\d{1,3}$/i.test(normalized);

  const justNo = normalized.match(/^(\d{1,3})$/);
  if (justNo) {
    const no = Number(justNo[1]);
    if (Number.isFinite(no) && no > 0) {
      out.add(`SK${no}`);
      if (car) out.add(`${car}_SK${no}`);
    }
  }

  const plainSkNo = normalized.match(/^SK(\d{1,3})$/);
  if (plainSkNo && car) {
    const no = Number(plainSkNo[1]);
    if (Number.isFinite(no) && no > 0) out.add(`${car}_SK${no}`);
  }

  if (car && !alreadyQualifiedSkinId) {
    const suffixNo = normalized.match(/(?:^|_)SK(\d{1,3})$/);
    if (suffixNo) {
      const no = Number(suffixNo[1]);
      if (Number.isFinite(no) && no > 0) out.add(`${car}_SK${no}`);
    }
  }
  return out;
}

function collectSkinOwnershipRawRows(input, depth = 0, out = []) {
  if (depth > 6 || input === null || input === undefined) return out;
  if (typeof input === "string" || typeof input === "number") {
    const text = String(input || "").trim();
    if (text) out.push(text);
    return out;
  }
  if (Array.isArray(input)) {
    for (const row of input) collectSkinOwnershipRawRows(row, depth + 1, out);
    return out;
  }
  if (typeof input !== "object") return out;
  const src = input;
  const directKeys = [
    "skinId",
    "SkinId",
    "SkinID",
    "skin",
    "Skin",
    "id",
    "ID",
    "key",
    "name",
    "Name",
    "value",
    "Value",
  ];
  for (const key of directKeys) {
    const text = String(src?.[key] || "").trim();
    if (text) out.push(text);
  }
  for (const [k, v] of Object.entries(src)) {
    const key = String(k || "").trim();
    if (key) out.push(key);
    if (/(skin|skins|owned|unlock|inventory|list|items|values?)/i.test(key)) {
      collectSkinOwnershipRawRows(v, depth + 1, out);
    }
  }
  return out;
}

const ownedSkinKeySet = computed(() => {
  const inventory = props.archiveGarageInventory && typeof props.archiveGarageInventory === "object" ? props.archiveGarageInventory : {};
  const source = Array.isArray(inventory?.skinsOwned) ? inventory.skinsOwned : [];
  const out = new Set();
  const carId = String(props.car?.id || "").trim().toUpperCase();
  for (const row of source) {
    for (const raw of collectSkinOwnershipRawRows(row)) {
      for (const key of expandSkinOwnershipKeys(raw, carId)) out.add(key);
    }
  }
  const cfg = getCurrentArchiveCarConfig();
  if (cfg && typeof cfg === "object") {
    const cfgBuckets = [
      cfg?.skinsOwned,
      cfg?.ownedSkins,
      cfg?.unlockedSkins,
      cfg?.skinIds,
      cfg?.skins,
      cfg?.skinMap,
    ];
    for (const raw of collectSkinOwnershipRawRows(cfgBuckets)) {
      for (const key of expandSkinOwnershipKeys(raw, carId)) out.add(key);
    }
  }
  return out;
});

function isSkinOwned(skin) {
  if (skin?.isDefault) return true;
  const carId = String(props.car?.id || "").trim().toUpperCase();

    .flatMap((v) => [...expandSkinOwnershipKeys(v, carId)])
    .filter(Boolean);
  return candidates.some((key) => ownedSkinKeySet.value.has(key));
}

function inferPartFromLookup(rawPart, slotType = "") {
  const src = rawPart && typeof rawPart === "object" ? rawPart : {};
  const idCandidates = [
    src?.id,
    src?.name,
    src?.partId,
    src?.sourcePartId,
    src?.DefID,
    src?.defId,
    src?.ID,
  ].map((v) => normalizePartIdForLookup(v)).filter(Boolean);
  for (const key of idCandidates) {
    if (PART_BY_ID.has(key)) return PART_BY_ID.get(key);
  }
  const pid = Number(src?.pathid);
  if (Number.isFinite(pid) && pid > 0 && PART_BY_PATHID.has(Math.floor(pid))) {
    return PART_BY_PATHID.get(Math.floor(pid));
  }
  const typeKey = String(src?.type || slotType || "").trim();
  const grade = String(src?.grade || "").trim();
  const variant = String(src?.variant || "").trim().toUpperCase();
  if (!typeKey || !PARTS_BY_TYPE[typeKey]) return null;
  const rows = Array.isArray(PARTS_BY_TYPE[typeKey]?.all) ? PARTS_BY_TYPE[typeKey].all : [];
  if (grade && variant) {
    const strict = rows.find((row) => String(row?.grade || "") === grade && String(row?.variant || "").toUpperCase() === variant);
    if (strict) return strict;
  }
  return rows[0] || null;
}

function resolvePartImagePath(part) {
  const src = part && typeof part === "object" ? part : {};
  const direct = String(src?.image || "").trim();
  if (direct && !/\/part-icons\/(?:Turbo|ECU|Damper|Tire)[A-Z]\w*\.png$/i.test(direct)) {
    return direct;
  }
  const typeKey = String(src?.type || "").trim();
  const effectRows = [
    ...(Array.isArray(src?.subEffects) ? src.subEffects : []),
    ...(Array.isArray(src?.effects) ? src.effects : []),
  ];
  const subKey = effectRows
    .map((row) => String(row?.key || "").trim())
    .find((key) => SUB_KEYS.includes(key));
  const mapped = LEGACY_PART_IMAGE_BY_TYPE_AND_KEY[typeKey]?.[subKey];
  if (mapped) return `/part-icons/${mapped}`;
  return direct;
}

function normalizedPartRecord(rawPart, slotType = "") {
  const src = rawPart && typeof rawPart === "object" ? rawPart : {};
  const canonical = inferPartFromLookup(src, slotType);
  const merged = canonical ? {
    ...canonical,
    ...src,
    effects: Array.isArray(src?.effects) && src.effects.length ? src.effects : canonical.effects,
    mainEffects: Array.isArray(src?.mainEffects) && src.mainEffects.length ? src.mainEffects : canonical.mainEffects,
    subEffects: Array.isArray(src?.subEffects) && src.subEffects.length ? src.subEffects : canonical.subEffects,
    breakthroughOptions: Array.isArray(src?.breakthroughOptions) && src.breakthroughOptions.length
      ? src.breakthroughOptions
      : canonical.breakthroughOptions,
  } : src;
  const image = resolvePartImagePath(merged);
  return image ? { ...merged, image } : merged;
}

function extractCarIdFromAssetUrl(rawUrl) {
  const text = String(rawUrl || "").trim();
  if (!text) return "";
  const skinMatch = text.match(/\/car-skins\/([^/_?]+)_sk\d+\.[^/?#]+(?:\?[^#]*)?$/i);
  if (skinMatch?.[1]) return String(skinMatch[1]).trim().toUpperCase();
  const spriteMatch = text.match(/\/car-sprites\/([^/.?]+)\.[^/?#]+(?:\?[^#]*)?$/i);
  if (spriteMatch?.[1]) return String(spriteMatch[1]).trim().toUpperCase();
  return "";
}

function replaceCarIdInAssetUrl(rawUrl, nextId) {
  const text = String(rawUrl || "").trim();
  const carId = String(nextId || "").trim().toUpperCase();
  if (!text || !carId) return text;
  let out = text;
  out = out.replace(/(\/car-skins\/)([^/_?]+)(_sk\d+\.[^/?#]+(?:\?[^#]*)?)$/i, `$1${encodeURIComponent(carId)}$3`);
  out = out.replace(/(\/car-sprites\/)([^/.?]+)(\.[^/?#]+(?:\?[^#]*)?)$/i, `$1${encodeURIComponent(carId)}$3`);
  return out;
}

function resolveCarIdAliasCandidates(...sources) {
  const queue = [];
  for (const raw of sources) {
    const text = String(raw || "").trim();
    if (!text) continue;
    const normalized = text.toUpperCase();
    queue.push(normalized);
    if (/_SK\d+$/i.test(normalized)) queue.push(normalized.replace(/_SK\d+$/i, ""));
  }
  const seen = new Set();
  const out = [];
  while (queue.length) {
    const id = String(queue.shift() || "").trim().toUpperCase();
    if (!id || seen.has(id)) continue;
    seen.add(id);
    out.push(id);
    const aliases = Array.isArray(CAR_ID_IMAGE_ALIASES[id]) ? CAR_ID_IMAGE_ALIASES[id] : [];
    for (const alias of aliases) queue.push(alias);
  }
  return out;
}

function buildAssetFallbackCandidates(rawUrl, ...idSources) {
  const primary = assetSrc(rawUrl);
  if (!primary) return [];
  const carIds = resolveCarIdAliasCandidates(extractCarIdFromAssetUrl(primary), ...idSources);
  const out = [primary];
  for (const id of carIds) {
    const replaced = replaceCarIdInAssetUrl(primary, id);
    if (replaced && !out.includes(replaced)) out.push(replaced);
  }
  return out;
}

function applyImageFallback(event, fallbackUrls, kind, note = "") {
  const target = event?.target;
  if (!(target instanceof HTMLImageElement)) {
    reportDetailAssetFailure(kind, event, note);
    return;
  }
  const rows = Array.isArray(fallbackUrls) ? fallbackUrls.filter(Boolean) : [];
  if (!rows.length) {
    reportDetailAssetFailure(kind, event, note);
    return;
  }
  const nextIndex = Number(target.dataset.wikiFallbackIndex || 0);
  if (nextIndex < rows.length) {
    target.dataset.wikiFallbackIndex = String(nextIndex + 1);
    target.src = rows[nextIndex];
    return;
  }
  reportDetailAssetFailure(kind, event, note);
}

function parseSkinBonusBlock(rawBonus) {
  const out = { absolute: zeroStats(), percent: zeroStats(), valid: false };
  if (!rawBonus || typeof rawBonus !== "object") return out;
  if (rawBonus.absolute || rawBonus.percent) {
    const absPart = rawBonus.absolute && typeof rawBonus.absolute === "object" ? rawBonus.absolute : {};
    const pctPart = rawBonus.percent && typeof rawBonus.percent === "object" ? rawBonus.percent : {};
    for (const k of ALL_KEYS) {
      out.absolute[k] = n(absPart[k]);
      let pct = n(pctPart[k]);
      if (Math.abs(pct) > 1) pct /= 100;
      out.percent[k] = pct;
    }
    out.valid = ALL_KEYS.some((k) => Math.abs(out.absolute[k]) > 1e-9 || Math.abs(out.percent[k]) > 1e-9);
    return out;
  }
  for (const k of ALL_KEYS) {
    let pct = n(rawBonus[k]);
    if (Math.abs(pct) > 1) pct /= 100;
    out.percent[k] = pct;
  }
  out.valid = ALL_KEYS.some((k) => Math.abs(out.percent[k]) > 1e-9);
  return out;
}

const skinContribution = computed(() => {
  const out = {
    absolute: zeroStats(),
    percent: zeroStats(),
    delta: zeroStats(),
    source: "none",
    sourceDesc: t("roster.detail.noSkinBonus"),
  };
  if (!includeSkinBonus.value) return out;

  const explicit = parseSkinBonusBlock(selectedSkin.value?.bonus);

      out.percent[k] = explicit.percent[k];
      out.delta[k] = explicit.absolute[k] + afterTier.value[k] * explicit.percent[k];
    }
    out.source = "skin_explicit";
    out.sourceDesc = t("roster.detail.skinApplied", { name: selectedSkinDisplayName() });
    return out;
  }

  const typeId = Number(props.car?.typeId ?? -1);
  const key = SKIN_FALLBACK_TYPE_KEY[typeId];
  if (!key) {
    out.sourceDesc = t("roster.detail.noSkinBonus");
    return out;
  }

  out.percent[key] = 0.06;
  out.delta[key] = afterTier.value[key] * 0.06;
  out.source = "default_type_fallback";
  out.sourceDesc = t("roster.detail.skinApplied", {
    name: selectedSkinDisplayName(),
  });
  return out;
});

const traitEffects = computed(() => {
  const rows = [];
  for (const trait of selectedTraits.value) {
    // Hidden bonuses are already merged into baseRaw via detail.hiddenBonus.
    // Do not apply them again as trait effects.
    if (isHiddenTrait(trait)) continue;
    const effects = Array.isArray(trait.effects) ? trait.effects : [];
    for (const rawEffect of effects) {
      const normalized = normalizeTraitEffect(rawEffect, trait);
      if (normalized) rows.push(normalized);
    }
  }
  return rows;
});

function resolveTraitFormulaZone(row) {
  const tierText = String(row?.unlockTier || "").toUpperCase();
  if (AS_MULTIPLY_TIERS.has(tierText)) return "as_multiply";
  if (SSS_ADDITIVE_TIERS.has(tierText)) return "sss_additive";
  return "ignored";
}

function panelReferenceValueForRequirement(key, alwaysAbs, alwaysPct, requirementAbs, requirementPct) {
  if (!USE_REQUESTED_PANEL_FORMULA) {
    const base = afterTier.value[key];
    const afterAlways = (base + alwaysAbs[key]) * (1 + alwaysPct[key]);
    const withSkin = afterAlways + skinContribution.value.delta[key];
    const withParts = withSkin + partsContribution.value.delta[key];
    return panelInt((withParts + requirementAbs[key]) * (1 + requirementPct[key]));
  }
  const base = afterTier.value[key];
  const afterAlways = (base + alwaysAbs[key]) * (1 + alwaysPct[key]);
  const withParts = afterAlways + partsContribution.value.delta[key];
  return panelInt((withParts + requirementAbs[key]) * (1 + requirementPct[key]));
}

const traitContribution = computed(() => {
  const out = {
    persistent: { absolute: zeroStats(), percent: zeroStats() },
    always: { absolute: zeroStats(), percent: zeroStats() },
    requirement: { absolute: zeroStats(), percent: zeroStats() },
    asMultiply: { absolute: zeroStats(), percent: zeroStats() },
    sssAdditive: { absolute: zeroStats(), percent: zeroStats() },
    ignored: { absolute: zeroStats(), percent: zeroStats() },
    conditional: { absolute: zeroStats(), percent: zeroStats(), notes: Object.fromEntries(ALL_KEYS.map((k) => [k, []])) },
    requirementStateByPathid: {},
    rows: [],
  };

  const applyPersistent = (row, bucket = "always") => {
    out.persistent.absolute[row.key] += row.absolute;
    out.persistent.percent[row.key] += row.percent;
    if (bucket === "always") {
      out.always.absolute[row.key] += row.absolute;
      out.always.percent[row.key] += row.percent;
    } else if (bucket === "requirement") {
      out.requirement.absolute[row.key] += row.absolute;
      out.requirement.percent[row.key] += row.percent;
    }

    const zone = resolveTraitFormulaZone(row);
    if (zone === "as_multiply") {
      out.asMultiply.absolute[row.key] += row.absolute;
      out.asMultiply.percent[row.key] += row.percent;
    } else if (zone === "sss_additive") {
      out.sssAdditive.absolute[row.key] += row.absolute;
      out.sssAdditive.percent[row.key] += row.percent;
    } else {
      out.ignored.absolute[row.key] += row.absolute;
      out.ignored.percent[row.key] += row.percent;
    }
    return zone;
  };

  const pushConditional = (row, note, conditionKind = "other") => {
    out.conditional.absolute[row.key] += row.absolute;
    out.conditional.percent[row.key] += row.percent;
    const text = String(note || row.conditionDesc || row.trigger || "").trim();
    if (text && !out.conditional.notes[row.key].includes(text)) out.conditional.notes[row.key].push(text);
    out.rows.push({
      ...row,
      formulaZone: "conditional",
      appliedInFormula: false,
      conditionKind,
      requirementSatisfied: false,
    });
  };

  const requirementRows = [];

  for (const row of traitEffects.value) {
    if (!row.hasCondition) {
      const zone = applyPersistent(row, "always");
      out.rows.push({
        ...row,
        formulaZone: zone,
        appliedInFormula: true,
        conditionKind: "none",
        requirementSatisfied: true,
      });
      continue;
    }

    const conditionText = String(row.conditionDesc || row.trigger || "").trim();
    const req = parseRequirementFromConditionText(conditionText);
    if (req) {
      requirementRows.push({ row, req, conditionText });
      continue;
    }
    if (isRuntimeOnlyConditionText(conditionText)) {
      pushConditional(row, `${conditionText}（局内生效，不计入面板）`, "runtime");
      continue;
    }
    pushConditional(row, conditionText || "条件未识别，按局内生效处理", "other");
  }

  const pending = requirementRows.slice();
  let progressed = true;
  while (pending.length && progressed) {
    progressed = false;
    for (let i = pending.length - 1; i >= 0; i -= 1) {
      const item = pending[i];
      const currentVal = panelReferenceValueForRequirement(
        item.req.key,
        out.always.absolute,
        out.always.percent,
        out.requirement.absolute,
        out.requirement.percent,
      );
      if (currentVal < item.req.threshold) continue;

      const zone = applyPersistent(item.row, "requirement");
      out.rows.push({
        ...item.row,
        formulaZone: zone,
        appliedInFormula: true,
        conditionKind: "requirement",
        requirementSatisfied: true,
        requirementKey: item.req.key,
        requirementThreshold: item.req.threshold,
      });
      out.requirementStateByPathid[item.row.pathid] = true;
      pending.splice(i, 1);
      progressed = true;
    }
  }

  for (const item of pending) {
    const currentVal = panelReferenceValueForRequirement(
      item.req.key,
      out.always.absolute,
      out.always.percent,
      out.requirement.absolute,
      out.requirement.percent,
    );
    const note = t("roster.detail.requirementCurrent", {
      condition: item.conditionText,
      label: metaLabel(item.req.key) || item.req.alias || item.req.key,
      value: currentVal,
    });
    pushConditional(item.row, note, "requirement");
    out.requirementStateByPathid[item.row.pathid] = false;
  }

  return out;
});

const finalPersistent = computed(() => {
  const out = zeroStats();
  if (!USE_REQUESTED_PANEL_FORMULA) {
    for (const k of ALL_KEYS) {
      const base = afterTier.value[k];
      const alwaysAbs = traitContribution.value.always.absolute[k];
      const alwaysPct = traitContribution.value.always.percent[k];
      const reqAbs = traitContribution.value.requirement.absolute[k];
      const reqPct = traitContribution.value.requirement.percent[k];
      const afterAlways = (base + alwaysAbs) * (1 + alwaysPct);
      const withSkin = afterAlways + skinContribution.value.delta[k];
      const withParts = withSkin + partsContribution.value.delta[k];
      out[k] = (withParts + reqAbs) * (1 + reqPct);
    }
    return out;
  }

  for (const k of ALL_KEYS) {
    const base = afterTier.value[k];
    const asAbs = traitContribution.value.asMultiply.absolute[k];
    const asPct = traitContribution.value.asMultiply.percent[k];
    const additiveAbs = skinContribution.value.absolute[k] + traitContribution.value.sssAdditive.absolute[k];
    const additivePct = skinContribution.value.percent[k] + traitContribution.value.sssAdditive.percent[k];
    const additiveDelta = additiveAbs + base * additivePct;
    out[k] = base * (1 + asPct) + asAbs + additiveDelta + partsContribution.value.delta[k];
  }
  return out;
});

const finalPotential = computed(() => {
  const out = zeroStats();
  for (const k of ALL_KEYS) {
    const base = finalPersistent.value[k];
    const abs = traitContribution.value.conditional.absolute[k];
    const pct = traitContribution.value.conditional.percent[k];
    out[k] = (base + abs) * (1 + pct);
  }
  return out;
});

function diffVsBase(key, val) {
  const base = baseRaw.value[key] || 0;
  const delta = val - base;
  const pct = base ? delta / base : 0;
  return {
    absolute: delta,
    percent: pct,
    text: `${fmtSigned(delta)} (${fmtSignedPct(pct)})`,
  };
}

function sourceLabel(item) {
  if (item.sourceType === "tier") return t("roster.detail.sourceLabels.tier");
  if (item.sourceType === "skin") return t("roster.detail.sourceLabels.skin");
  if (item.sourceType === "trait_as") return t("roster.detail.sourceLabels.traitAs");
  if (item.sourceType === "trait_sss") return t("roster.detail.sourceLabels.traitSss");
  if (item.sourceType === "trait") return item.conditional ? t("roster.detail.sourceLabels.traitConditional") : t("roster.detail.sourceLabels.traitPersistent");
  return item.sourceType || t("roster.detail.sourceLabels.default");
}

const affixRows = computed(() => {

  for (const k of ALL_KEYS) {
    const abs = skinContribution.value.absolute[k];
    const pct = skinContribution.value.percent[k];

        percent: pct,
        conditional: false,
        source: skinContribution.value.sourceDesc,
        sourceType: "skin",
      });

      description: tr.descriptionZh || tr.conditionDesc || tr.trigger,
      sourceType: traitSourceType,
    });
  }

  const decorate = (row) => ({ ...row, sourceText: sourceLabel(row) });
  return {
    main: rows.filter((r) => MAIN_KEYS.includes(r.key)).map(decorate),
    sub: rows.filter((r) => SUB_KEYS.includes(r.key)).map(decorate),
  };

const traitPanelStateByPathid = computed(() => {
  const map = {};
  for (const row of traitContribution.value.rows) {
    const pid = normalizeTraitPathid(row.pathid);
    if (!pid) continue;
    const prev = map[pid] || { runtime: false, requirement: false, requirementSatisfied: null };
    if (row.conditionKind === "runtime") prev.runtime = true;
    if (row.conditionKind === "requirement") {
      prev.requirement = true;
      if (prev.requirementSatisfied === null) prev.requirementSatisfied = Boolean(row.requirementSatisfied);
      else prev.requirementSatisfied = prev.requirementSatisfied && Boolean(row.requirementSatisfied);
    }
    map[pid] = prev;
  }
  return map;
});

      absolute: skinRows.reduce((s, r) => s + n(r.computed), 0),

    },
    rows: skinRows,
  });

  const partRows = ALL_KEYS.filter((k) => Math.abs(partsContribution.value.delta[k]) > 1e-9).map((k) => {
    const sources = [];
    for (const eq of Object.values(equippedParts.value)) {
      if (!eq) continue;
      const partName = String(eq.part?.displayNameZh || eq.part?.name || t("roster.detail.partFallbackName"));
      if (Array.isArray(eq.part?.effects) && eq.part.effects.some((e) => String(e?.key || "") === k)) sources.push(partName);
      const btRows = Array.isArray(eq.part?.breakthroughOptions)
        ? (Array.isArray(eq.customBreakthroughPathids) ? eq.customBreakthroughPathids : [])
            .map((v) => Number(v))
            .filter((v) => Number.isFinite(v) && v > 0)
            .map((pid) => eq.part.breakthroughOptions.find((o) => Number(o.pathid) === pid))
            .filter(Boolean)
        : [];
      for (const bt of btRows) {
        if (String(bt.key || "") === k) sources.push(`${partName}(${t("roster.detail.breakthroughShort")}:${bt.labelZh || metaLabel(k) || k})`);
      }
    }

      key: k,
      label: metaLabel(k),
      absolute: partsContribution.value.absolute[k],
      percent: partsContribution.value.percent[k],
      computed: partsContribution.value.delta[k],
      source: "parts",
      note: sources.join(", ") || t("roster.detail.partsEquipped"),
    };
  });
  groups.push({
    key: "parts",
    title: breakdownGroupTitle("parts"),
    summary: sumRows(partRows),
    rows: partRows,
  });

  const persistentRows = traitContribution.value.rows
    .filter((r) => Boolean(r.appliedInFormula))
    .map((r) => ({
      key: r.key,
      label: metaLabel(r.key) || r.label,
      absolute: r.absolute,
      percent: r.percent,
      computed: afterTier.value[r.key] * r.percent + r.absolute,
      source: "trait_persistent",
      note: `${r.source} (${tierLabel(r.unlockTier || "-")})${r.conditionKind === "requirement" ? ` / ${t("roster.detail.requirementMetShort")}` : ""}${USE_REQUESTED_PANEL_FORMULA ? (r.formulaZone === "as_multiply" ? ` / ${t("roster.detail.sourceLabels.traitAs")}` : ` / ${t("roster.detail.sourceLabels.traitSss")}`) : ""}`,
      description: r.descriptionZh || r.trigger,
    }));
  groups.push({
    key: "trait_persistent",
    title: breakdownGroupTitle("trait_persistent"),
    summary: sumRows(persistentRows),
    rows: persistentRows,
  });

  const conditionRows = traitContribution.value.rows
    .filter((r) => r.hasCondition && !r.appliedInFormula)
    .map((r) => ({
      key: r.key,
      label: metaLabel(r.key) || r.label,
      absolute: r.absolute,
      percent: r.percent,
      computed: finalPersistent.value[r.key] * r.percent + r.absolute,
      source: "trait_conditional",
      note: `${r.source} / ${r.conditionDesc || r.trigger}`,
      trigger: r.conditionDesc || r.trigger,
      description: r.descriptionZh || r.conditionDesc || r.trigger,
    }));
  groups.push({
    key: "trait_conditional",
    title: breakdownGroupTitle("trait_conditional"),
    summary: sumRows(conditionRows),
    rows: conditionRows,
  });

  return groups;
});

function maxTierOption() {
  if (!availableTierOptions.value.length) return "S";
  const optionSet = new Set(availableTierOptions.value.map((t) => String(t || "").toUpperCase()));
  for (let i = TIER_ORDER.length - 1; i >= 0; i -= 1) {
    const tierName = TIER_ORDER[i];
    if (optionSet.has(tierName)) return tierName;
  }
  return String(availableTierOptions.value[availableTierOptions.value.length - 1] || "S").toUpperCase();
}

function normalizeTierName(rawTier) {
  const t = String(rawTier || "").trim().toUpperCase();
  if (!t) return "";
  if (TIER_ORDER.includes(t)) return t;
  if (t.includes("SSS")) return "SSS";
  if (t.includes("SS")) return "SS";
  if (t.includes("S")) return "S";
  if (t.includes("A")) return "A";
  if (t.includes("B")) return "B";
  if (t.includes("C")) return "C";
  return "";
}

function collectTraitPathidsForTier(targetTier) {
  const idx = Math.max(0, TIER_ORDER.indexOf(normalizeTierName(targetTier || "S") || "S"));

  return allTraits.value
    .filter((t) => {

      return allow.has(normalizeTierName(t.unlockTier || t.unlock_tier));

    .filter(Boolean);
}

function effectTextFromTrait(trait) {
  const effects = Array.isArray(trait.effects) ? trait.effects : [];
  return effects
    .map((e) => {
      const key = String(e.key || "");
      const label = String(e.type_zh || metaLabel(key) || "").trim();
      if (!label) return "";
      const abs = n(e.absolute);
      const pct = n(e.percent);
      if (Math.abs(pct) > 1e-9) return `${label}${fmtSignedPct(pct)}`;
      return `${label}${fmtSigned(abs)}`;
    })
    .filter(Boolean)
    .join("；");
}

function allMainSamePercentSummary(trait) {
  const effects = Array.isArray(trait.effects) ? trait.effects : [];
  const mainRows = effects.filter((e) => MAIN_KEYS.includes(String(e?.key || "")));
  if (mainRows.length < 4) return "";
  const byKey = new Map(mainRows.map((e) => [String(e.key), e]));
  if (!MAIN_KEYS.every((k) => byKey.has(k))) return "";
  const rows = MAIN_KEYS.map((k) => byKey.get(k));
  if (!rows.every((e) => Math.abs(n(e.absolute)) < 1e-9)) return "";
  const p0 = n(rows[0].percent);
  if (Math.abs(p0) < 1e-9) return "";
  if (!rows.every((e) => Math.abs(n(e.percent) - p0) < 1e-9)) return "";
  return `主要属性${fmtSignedPct(p0)}`;
}

function displayTraitDesc(trait) {
  const explicitDesc = String(
    locale.value === "en_us"
      ? (trait?.descriptionEn || trait?.description_en || "")
      : (trait?.descriptionZh || trait?.description_zh || ""),
  ).trim();
  if (explicitDesc && !/Limit\s*=|Param1\s*=|常驻\/条件见说明|Always active \/ see condition note/.test(explicitDesc)) return explicitDesc;

  const trigger = String(
    locale.value === "en_us"
      ? (trait?.triggerEn || trait?.trigger_en || trait?.triggerText || trait?.trigger || "")
      : (trait?.triggerText || trait?.trigger || ""),
  ).trim();
  const effectsText = String(
    locale.value === "en_us"
      ? (trait?.effectsEn || trait?.effects_en || "")
      : (trait?.effectsZh || trait?.effects_zh || ""),
  ).trim();
  if (trigger && effectsText) return `${trigger}${locale.value === "en_us" ? ", " : "，"}${effectsText.split(";")[0]?.trim() || effectsText}`;
  if (effectsText) return effectsText;
  if (trigger) return trigger;

  const allMainSummary = allMainSamePercentSummary(trait);
  const fromHidden = conditionTextFromLimit(
    parseHiddenValues(trait.hiddenValues || trait.hidden_values)?.limit,
    parseHiddenValues(trait.hiddenValues || trait.hidden_values)?.param1,
  );
  if (fromHidden) {
    const effect = allMainSummary || effectTextFromTrait(trait);
    return effect ? `${fromHidden}，${effect}` : fromHidden;
  }
  if (allMainSummary) {
    const zhTrigger = String(trait.triggerText || trait.trigger || "").trim();
    if (zhTrigger && !/Limit\s*=|Param1\s*=|常驻\/条件见说明/.test(zhTrigger)) {
      return `${zhTrigger}，${allMainSummary}`;
    }
    return allMainSummary;
  }
  const desc = String(trait.descriptionZh || trait.description_zh || "").trim();
  if (desc && !/Limit\s*=|Param1\s*=|常驻\/条件见说明/.test(desc)) return desc;
  const zhTrigger = String(trait.triggerText || trait.trigger || "").trim();
  if (zhTrigger && !/Limit\s*=|Param1\s*=|常驻\/条件见说明/.test(zhTrigger)) return zhTrigger;
  return effectTextFromTrait(trait) || t("roster.detail.noDescription");
}

function onToggleTrait(pathid) {
  if (isVisitorReadOnly.value) return;
  const pid = normalizeTraitPathid(pathid);
  if (!pid) return;
  const target = unlockedTraits.value.find((t) => normalizeTraitPathid(t.pathid) === pid);
  if (!target) return;
  const isHidden = String(target.source || "") === "HiddenUpgrades";
  if (isHidden) return;

  const idx = selectedTraitPathids.value.findIndex((v) => normalizeTraitPathid(v) === pid);
  if (idx >= 0) {
    selectedTraitPathids.value = selectedTraitPathids.value.filter((v) => normalizeTraitPathid(v) !== pid);
    return;
  }

  // Traits are multi-select by design.
  selectedTraitPathids.value = [...selectedTraitPathids.value, pid];
}

function isCustomizableSPart(part) {
  if (String(part?.grade || "") !== "S") return false;
  return Array.isArray(part?.breakthroughOptions) && part.breakthroughOptions.length > 0;
}
function breakthroughOptionsForPart(part) {
  if (!isCustomizableSPart(part)) return [];
  const ownRows = normalizeOptionRows(part?.breakthroughOptions);
  const typeKey = String(part?.type || "");
  const allowedKeys = TYPE_BREAKTHROUGH_ALLOWED_KEYS[typeKey];
  const filtered = allowedKeys instanceof Set ? ownRows.filter((r) => allowedKeys.has(String(r?.key || ""))) : ownRows;
  return dedupeBreakthroughOptions(filtered);
}
function findBreakthroughOption(part, pathid) {
  const pid = Number(pathid);
  if (!Number.isFinite(pid) || pid <= 0) return null;
  return breakthroughOptionsForPart(part).find((o) => Number(o.pathid) === pid) || null;
}
function normalizeBreakthroughLevel(v) {
  const nlv = Math.floor(Number(v) || 1);
  return Math.max(1, Math.min(4, nlv));
}
function breakthroughMultiplier(level) {
  return normalizeBreakthroughLevel(level) / 4;
}
function scaledBreakthroughOption(opt, level) {
  const mul = breakthroughMultiplier(level);
  return {
    ...opt,
    absolute: n(opt?.absolute) * mul,
    percent: n(opt?.percent) * mul,
    baseAbsolute: n(opt?.absolute),
    basePercent: n(opt?.percent),
    breakthroughLevel: normalizeBreakthroughLevel(level),
  };
}
function isReadonlyImportedBreakthroughRow(row) {
  return Boolean(row?.isImportedOnly || row?.isDuplicateSelection);
}
function normalizeImportedBreakthroughRow(rawRow, eqData) {
  const src = rawRow && typeof rawRow === "object" ? rawRow : {};
  const pid = Number(src?.pathid);
  if (!Number.isFinite(pid) || pid <= 0) return null;
  const fallbackOpt = findBreakthroughOption(eqData?.part, pid);
  const base = fallbackOpt && !src?.key ? { ...fallbackOpt, ...src } : src;
  const key = String(base?.key || "").trim();
  if (!ALL_KEYS.includes(key)) return null;
  const level = normalizeBreakthroughLevel(src?.level ?? src?.breakthroughLevel ?? 1);
  return {
    ...scaledBreakthroughOption(base, level),
    modifierName: String(base?.modifierName || base?.name || "").trim(),
    isNative: Boolean(base?.isNative),
    isImportedOnly: Boolean(base?.isImportedOnly),
    isDuplicateSelection: Boolean(base?.isDuplicateSelection),
    sourcePartId: String(base?.sourcePartId || "").trim(),
    sourcePartType: String(base?.sourcePartType || "").trim(),
  };
}
function selectedBreakthroughRows(eqData) {
  if (!eqData || !isCustomizableSPart(eqData?.part)) return [];
  const importedRows = Array.isArray(eqData?.importedBreakthroughRows) ? eqData.importedBreakthroughRows : [];
  if (importedRows.length) {
    return importedRows
      .map((row) => normalizeImportedBreakthroughRow(row, eqData))
      .filter(Boolean);
  }
  const idsRaw = Array.isArray(eqData?.customBreakthroughPathids) ? eqData.customBreakthroughPathids : [];
  const levelsRaw = Array.isArray(eqData?.breakthroughLevels)
    ? eqData.breakthroughLevels
    : [eqData?.breakthroughLevel, eqData?.breakthroughLevel];
  const out = [];
  for (let i = 0; i < idsRaw.length; i += 1) {
    const pid = Number(idsRaw[i]);
    if (!Number.isFinite(pid) || pid <= 0) continue;
    const opt = findBreakthroughOption(eqData.part, pid);
    if (!opt) continue;
    const lv = normalizeBreakthroughLevel(levelsRaw[i] ?? eqData?.breakthroughLevel ?? 1);
    out.push(scaledBreakthroughOption(opt, lv));
  }
  return out;
}

      const g = (gradeDesc[String(a?.grade || "")] ?? 99) - (gradeDesc[String(b?.grade || "")] ?? 99);
      if (g) return g;
      const v = String(a?.variant || "").localeCompare(String(b?.variant || ""));

  if (!config || typeof config !== "object") return;
  if (config.tier) tier.value = String(config.tier);
  if (Number.isFinite(Number(config.level))) level.value = Number(config.level);
  includeSkinBonus.value = Boolean(config.includeSkinBonus);
  selectedSkinId.value = String(config.selectedSkinId || selectedSkinId.value || "");
  selectedTraitPathids.value = Array.isArray(config.selectedTraitPathids)
    ? config.selectedTraitPathids.map((v) => normalizeTraitPathid(v)).filter(Boolean)
    : [];
  if (config.equippedParts && typeof config.equippedParts === "object") {
    equippedParts.value = normalizeArchiveEquippedParts(JSON.parse(JSON.stringify(config.equippedParts)));
  }
}

async function loadCloudConfig() {
  const token = String(props.cloudToken || "");
  const carId = String(props.car?.id || "").toUpperCase();
  if (!token || !carId) return;
  cloudBusy.value = true;
  try {
    const json = await cloudApi.getCarConfig(token, carId);
    if (json?.config) {
      applyCloudConfig(json.config);
      cloudMsg.value = t("roster.detail.cloudMessages.loadedConfig");
    } else {
      cloudMsg.value = t("roster.detail.cloudMessages.noConfig");
    }
  } catch (err) {
    cloudMsg.value = t("roster.detail.cloudMessages.loadFailed", { message: err?.message || "Unknown error" });
  } finally {
    cloudBusy.value = false;
  }
}

async function saveCloudConfig() {
  const token = String(props.cloudToken || "");
  const carId = String(props.car?.id || "").toUpperCase();
  if (!token || !carId) {
    cloudMsg.value = t("roster.detail.cloudMessages.loginRequired");
    return;
  }
  cloudBusy.value = true;
  try {
    const payload = buildCloudConfig();
    await cloudApi.saveCarConfig(token, carId, payload);
    cloudMsg.value = t("roster.detail.cloudMessages.savedConfig");
  } catch (err) {
    cloudMsg.value = t("roster.detail.cloudMessages.saveFailed", { message: err?.message || "Unknown error" });
  } finally {
    cloudBusy.value = false;
  }
}

async function loadCloudPresets() {
  const token = String(props.cloudToken || "");
  const carId = String(props.car?.id || "").toUpperCase();
  if (!token) {
    cloudPresets.value = [];
    selectedCloudPresetId.value = "";
    return;
  }
  try {
    const json = await cloudApi.getPresets(token, carId);
    cloudPresets.value = Array.isArray(json?.presets) ? json.presets : [];
    if (!cloudPresets.value.some((p) => String(p.id) === String(selectedCloudPresetId.value))) {
      selectedCloudPresetId.value = cloudPresets.value[0] ? String(cloudPresets.value[0].id) : "";
    }
  } catch {
    cloudPresets.value = [];
  }
}

async function saveAsCloudPreset() {
  const token = String(props.cloudToken || "");
  const carId = String(props.car?.id || "").toUpperCase();
  if (!token) {
    cloudMsg.value = t("roster.detail.cloudMessages.loginRequired");
    return;
  }
  if (!carId) {
    cloudMsg.value = t("roster.detail.cloudMessages.invalidCarId");
    return;
  }
  if (cloudPresets.value.length >= MAX_PRESETS_PER_CAR) {
    cloudMsg.value = t("roster.detail.cloudMessages.presetLimit", { max: MAX_PRESETS_PER_CAR });
    return;
  }
  cloudBusy.value = true;
  try {
    const baseName = locale.value === "en_us"
      ? (carNameEn.value || carPrimaryName.value || props.car?.id || "Car")
      : (carNameZh.value || carPrimaryName.value || props.car?.id || "赛车");
    const fallbackName = locale.value === "en_us"
      ? t("roster.detail.cloudMessages.presetNameDefaultEn", { car: baseName })
      : t("roster.detail.cloudMessages.presetNameDefaultZh", { car: baseName });
    const name = String(cloudPresetName.value || "").trim() || fallbackName;
    const data = buildCloudConfig();
    await cloudApi.createPreset(token, { name, data, carId });
    cloudPresetName.value = "";
    await loadCloudPresets();
    cloudMsg.value = t("roster.detail.cloudMessages.presetSaved");
  } catch (err) {
    cloudMsg.value = t("roster.detail.cloudMessages.presetSaveFailed", { message: err?.message || "Unknown error" });
  } finally {
    cloudBusy.value = false;
  }
}

async function removeCloudPreset(presetId) {

  try {

  } catch (err) {
    cloudMsg.value = t("roster.detail.cloudMessages.presetDeleteFailed", { message: err?.message || "Unknown error" });
  } finally {
    cloudBusy.value = false;
  }
}

function parseGarageCarLevel(garage, carId) {
  const normalizedId = String(carId || "").trim().toUpperCase();
  if (!normalizedId) return 0;
  const carsMap = garage && typeof garage === "object" ? garage : {};
  const direct = carsMap?.[normalizedId] || carsMap?.[String(carId || "").trim()];
  if (direct && typeof direct === "object") {
    const lv = Number(direct.level);
    if (Number.isFinite(lv) && lv > 0) return lv;
  }
  return 0;
}

function parseGarageCarUpgradeStage(garage, carId) {
  const normalizedId = String(carId || "").trim().toUpperCase();
  if (!normalizedId) return 0;
  const carsMap = garage && typeof garage === "object" ? garage : {};
  const direct = carsMap?.[normalizedId] || carsMap?.[String(carId || "").trim()];
  if (direct && typeof direct === "object") {
    const stage = Number(direct.upgradeStage);
    if (Number.isFinite(stage) && stage >= 0) return Math.floor(stage);
  }
  return 0;
}

function traitPathidsFromUpgradeStage(stageRaw) {
  const stage = Number(stageRaw);
  if (!Number.isFinite(stage) || stage <= 0) return [];
  const maxTierIdx = Math.max(0, Math.min(TIER_ORDER.length - 1, Math.floor(stage) - 1));
  const out = [];
  const rows = Array.isArray(props.detail?.traits) ? props.detail.traits : [];
  for (const tr of rows) {
    const pid = normalizeTraitPathid(tr?.pathid);
    if (!pid) continue;
    const src = String(tr?.source || "");
    if (src === "HiddenUpgrades") {
      out.push(pid);
      continue;
    }
    if (src !== "Upgrades") continue;
    const unlockTier = normalizeTierName(tr?.unlockTier || tr?.unlock_tier);
    const unlockIdx = TIER_ORDER.indexOf(unlockTier);
    if (unlockIdx < 0) continue;
    if (unlockIdx <= maxTierIdx) out.push(pid);
  }
  return Array.from(new Set(out));
}

function normalizeArchiveEquippedParts(parts) {
  const input = parts && typeof parts === "object" ? parts : {};
  const bySlot = {
    turbo: input.turbo ?? null,
    ecu: input.ecu ?? null,
    suspension: input.suspension ?? null,
    tires: input.tires ?? null,
  };
  for (const slot of Object.keys(bySlot)) {
    const eq = bySlot[slot];
    if (!eq || typeof eq !== "object") continue;
    const partRaw = eq?.part && typeof eq.part === "object" ? eq.part : eq;
    const part = normalizedPartRecord(partRaw, slot);
    const levelsRaw = Array.isArray(eq.breakthroughLevels) ? eq.breakthroughLevels : [eq.breakthroughLevel, eq.breakthroughLevel];
    const levels = [0, 1].map((idx) => {
      const v = Number(levelsRaw[idx] ?? levelsRaw[0] ?? 1);
      return Number.isFinite(v) && v > 0 ? Math.floor(v) : 1;
    });
    bySlot[slot] = {
      ...eq,
      part,
      breakthroughLevels: levels,
      breakthroughLevel: levels[0],
    };
  }
  return bySlot;
}

function resolveArchiveSkinForCurrentCar(rawSkinId) {
  const wanted = String(rawSkinId || "").trim();
  if (!wanted) return { skinId: "", isDefault: false };
  const list = Array.isArray(skins.value) ? skins.value : [];
  if (!list.length) return { skinId: "", isDefault: false };

  const exact = list.find((s) => String(s?.id || "") === wanted);
  if (exact) return { skinId: String(exact.id), isDefault: Boolean(exact?.isDefault) };

  const wantedUpper = wanted.toUpperCase();
  const ci = list.find((s) => String(s?.id || "").toUpperCase() === wantedUpper);
  if (ci) return { skinId: String(ci.id), isDefault: Boolean(ci?.isDefault) };

  const skNoMatch = wantedUpper.match(/_SK(\d+)$/);
  if (skNoMatch) {
    const skNo = Number(skNoMatch[1]);
    if (Number.isFinite(skNo) && skNo >= 0) {
      const suffix = `_SK${Math.floor(skNo)}`;
      const bySuffix = list.find((s) => String(s?.id || "").toUpperCase().endsWith(suffix));
      if (bySuffix) return { skinId: String(bySuffix.id), isDefault: Boolean(bySuffix?.isDefault) };
    }
  }

  return { skinId: "", isDefault: false };
}

function applyArchiveCarConfig(carCfg) {
  if (!carCfg || typeof carCfg !== "object") return;
  const lv = Number(carCfg.level);
  if (Number.isFinite(lv) && lv > 0) level.value = Math.max(1, Math.min(30, Math.floor(lv)));
  const targetTier = String(carCfg.tier || "").toUpperCase();
  if (targetTier && availableTierOptions.value.includes(targetTier)) tier.value = targetTier;
  let skinBonusResolved = false;
  if (typeof carCfg.includeSkinBonus === "boolean") {
    includeSkinBonus.value = carCfg.includeSkinBonus;
    skinBonusResolved = true;
  }
  const skinId = String(carCfg.selectedSkinId || "").trim();
  if (skinId) {
    const picked = resolveArchiveSkinForCurrentCar(skinId);
    if (picked.skinId) {
      selectedSkinId.value = picked.skinId;
      if (!skinBonusResolved) includeSkinBonus.value = !picked.isDefault;
      skinBonusResolved = true;
    } else if (!skinBonusResolved) {
      // Archive says an equipped skin exists, but this client cannot map that skin id.
      // Keep bonus enabled while UI falls back to default skin card.
      includeSkinBonus.value = true;
      skinBonusResolved = true;
    }
  }
  if (!skinBonusResolved && !selectedSkin.value?.isDefault) {
    includeSkinBonus.value = true;
  }
  const traitRows = Array.isArray(carCfg.selectedTraitPathids) ? carCfg.selectedTraitPathids : [];
  if (traitRows.length) {
    selectedTraitPathids.value = traitRows
      .map((v) => normalizeTraitPathid(v))
      .filter(Boolean);
  }
  const parts = carCfg.equippedParts && typeof carCfg.equippedParts === "object" ? carCfg.equippedParts : null;
  if (parts) {
    equippedParts.value = normalizeArchiveEquippedParts(parts);
  }
}

async function syncGarageToDetailStub(options = {}) {
  const silent = Boolean(options?.silent);
  const carId = String(props.car?.id || "").trim().toUpperCase();
  if (!props.archiveReadEnabled) {
    if (!silent) cloudMsg.value = t("roster.detail.cloudMessages.enableArchiveReadFirst");
    return false;
  }
  if (!carId) {
    if (!silent) cloudMsg.value = t("roster.detail.cloudMessages.invalidCarId");
    return false;
  }
  const garageMap = props.archiveGarageState && typeof props.archiveGarageState === "object" ? props.archiveGarageState : {};
  const carCfg = getCurrentArchiveCarConfig();
  const levelFromGarage = parseGarageCarLevel(garageMap, carId);
  const upgradeStageFromGarage = parseGarageCarUpgradeStage(garageMap, carId);
  if (!levelFromGarage) {
    if (!silent) cloudMsg.value = t("roster.detail.cloudMessages.archiveMissingCar");
    return false;
  }
  level.value = Math.max(1, Math.min(30, levelFromGarage));
  applyArchiveCarConfig(carCfg);
  // Cloud payload only gives upgrade stage (1=C ... 6=SSS), not explicit per-trait toggles.
  // Apply stage-based trait selection so TRAITS MODULE reflects archive progression.
  const stageTraitPathids = traitPathidsFromUpgradeStage(upgradeStageFromGarage);
  const hiddenTraitPathids = (Array.isArray(props.detail?.traits) ? props.detail.traits : [])
    .filter((tr) => String(tr?.source || "") === "HiddenUpgrades")
    .map((tr) => normalizeTraitPathid(tr?.pathid))
    .filter(Boolean);
  const mergedTraitPathids = Array.from(new Set([...stageTraitPathids, ...hiddenTraitPathids]));
  selectedTraitPathids.value = mergedTraitPathids;
  traitInitCarId.value = String(carId || "");
  archiveAppliedCarId.value = String(carId || "");
  cloudMsg.value = t("roster.detail.cloudMessages.archiveSynced", {
    level: levelFromGarage,
    stage: upgradeStageFromGarage,
    count: mergedTraitPathids.length,
    source: isSelfArchiveData.value ? t("common.actions.self") : t("roster.detail.previewMode"),
  });
  return true;
}
